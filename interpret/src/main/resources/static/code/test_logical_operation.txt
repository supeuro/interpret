bool logicalOperation_test()
{
	int a = 1;
	bool b = false;
	if (!a)
	{
		return false;
	}
	if (a && b)
	{
		return false;
	}
	if (a && b || b)
	{
		return false;
	}
	if (b || b && a)
	{
		return false;
	}

	if ((a = 1) || (b = 1))
	{
		b = 1;
		b = 0;
	}
	if ((a = 0) && (b = 1))
	{
		return false;
	}

	if (a || b)
	{
		return true;
	}
}
