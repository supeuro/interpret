bool logicalOperation_test()
{

		return false;
	}

	{
		return false;
	}
	if (a && b || b)
	{
		return false;
	}
	if (b || b && a)
	{
		return false;
	}

	{
		b = 1;
		b = 0;
	}

	}

	{
		return true;
